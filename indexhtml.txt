 function sortTable() {
  var table, rows, sorting, i, x, y, change;
  table = document.getElementById("fumTable");
  sorting = true;
  
  while (sorting) {
   
    sorting = false;
    rows = table.rows;
   
    for (i = 1; i < (rows.length - 1); i++) {
       
      change = false;
   
      x = rows[i].getElementsByTagName("TD")[0];
      y = rows[i + 1].getElementsByTagName("TD")[0];
     
      if (y.innerHTML.toUpperCase() < x.innerHTML.toUpperCase()) {
        
        change = true;
        break;
      }
    }
    if (change) {
       
      rows[i].parentNode.insertBefore(rows[i + 1], rows[i]);
      sorting = true;
    }
  }
}
